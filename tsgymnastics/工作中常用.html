模式匹配 #

typescript
// 数组类型
// First
type GetFirst<T extends unknown[]> = T extends [infer V, ...unknown[]] ? V : never

// Last
type GetLast<T extends unknown[]> = T extends [...unknown[], infer V] ? V : never

// PopArr
type PopArr<T extends unknown[]> = T extends [] ? [] : T extends [...infer Rest, unknown/infer] ? Rest : never

// ShiftArr
type ShiftArr<T extends unknown[]> = T extends [] ? [] : T extends [unknown, ...infer V] ? V : never


// 字符串类型
// StartsWith：判断字符串是否以某个 前缀开头
type StartsWith<T extends string, P extends string> = T extends `${P}${string}` ? true : false

// ReplaceStr：匹配第一个满足条件的字符
type ReplaceStr<Str extends string, From extends string, To extends string> = 
Str extends `${infer Prefix}${From}${infer Suffix}` ? `${Prefix}${To}${Suffix}` : Str
type ReplaceAll<Str extends string, From extends string, To extends string> = 
Str extends `${infer Prefix}${From}${infer Suffix}` ? ReplaceAll<`${Prefix}${To}${Suffix}`, From, To> : Str

// Trim
// TrimRight
type TrimStrRight<Str extends string> = Str extends `${infer Rest}${' ' | '\n' | '\t'}` ? TrimStrRight<Rest> : Str
// TrimLeft
type TrimStrLeft<Str extends string> = Str extends `${' ' | '\n' | '\t'}${infer Rest}` ? TrimStrLeft<Rest> : Str
//TrimStr
type TrimStr<Str extends string> = TrimStrRight<TrimStrLeft<Str>>


// 函数
// 提取函数参数
type GetParameters<Func extends Function> = Func extends (...args: infer Args) => unknown ? Args : never;

// 提取返回值类型
// any 是既可以接收类型，也可以赋值给其他类型，而 unknown 只能接收类型。函数参数是逆变，返回值是协变
type GetReturnType<Func extends Function> = Func extends (...args: any[]) => infer ReturnType ? ReturnType : never;

interface Person {

type GetConstructorParameters<
    ConstructorType extends new (...args: any) => any
> = ConstructorType extends new (...args: infer ParametersType) => any
    ? ParametersType
    : never;

    'ref' extends keyof Props
        ? Props extends { ref?: infer Value | undefined}
            ? Value
            : never
        : never;

重新构造 #

  • 对索引做修改的as叫做重映射
  • Record<string, any>：内置高级类型，不能是unknow，因为函数参数要赋值给其他类
typescript
// 数组类型的重新构造
// push 往数组/元组尾部添加类型
type Push<Arr extends unknown[], Ele> = [...Arr, Ele]
type Res = Push<[1, 2, 3], 4>

// unshift 往数组/元组头部添加类型
type UnShift<Arr extends unknown[], Ele> = [Ele, ...Arr]

// Zip 合并数组
// 有限个
type Zip<T extends unknown[], R extends unknown[]> = T extends [infer TOne, infer TTwo] ? R extends [infer ROne, infer RTwo] ? [[TOne, ROne], [TTwo, RTwo]] : [] : []
// 任意：递归
type Zip<T extends unknown[], R extends unknown[]> = T extends [infer TOne, ...infer TRest] ? R extends [infer ROne, ...infer RRest] ? [[TOne, ROne], ...Zip<TRest, RRest>] : [] : []
type tuple = Zip<[1, 2], ['guang', 'dong']>

// 字符串类型的重新构造
// CapitalizeStr 首字母大写 Uppercase 把首字母转为大写(内置高级类型)
type CapitalizeStr<Str extends string> = Str extends `${infer First}${infer Rest}` ? `${Uppercase<First>}${Rest}` : Str
type res = CapitalizeStr<'absc'>

//  dong_dong_dong 到 dongDongDong 的变换。
type CamelCase<Str extends string> = Str extends `${infer Left}_${infer Right}${infer Rest}` ?
    `${Left}${Uppercase<Right>}${CamelCase<Rest>}` : Str
type res = CamelCase<'dong_dong_dong'>

// DropSubStr 删除字符串中的某个字符串
type DropSubStr<Str extends string, sym extends string> = Str extends `${infer Left}${sym}${infer Right}` ? DropSubStr<`${Left}${Right}`, sym> : Str
type res = DropSubStr<'dong~~~', '~'>


// 函数类型的重新构造
// AppendArgument 实现参数添加
type AppendArgument<Func extends Function, Arg> = Func extends (...args: infer Args) => infer ReturnType ? 
    (...args: [...Args, Arg]) => ReturnType : never
type res = AppendArgument<(name: string) => boolean, number>


// 索引类型的重新构造
// Mapping 构造三个元素的元组类型
type Mapping<Obj extends object> = {
    [Key in keyof Obj]: [Obj[Key], Obj[Key], Obj[Key]]
}
type res = Mapping<{a: 1, b: 2}>

// UppercaseKey 把索引类型的Key变为大写
type UppercaseKey<Obj extends object> = {
    [Key in keyof Obj as Uppercase<Key & string>]: Obj[Key]
}

// Record 创建索引类型（内置的高级类型）
type UppercaseKey<Obj extends Record<string, any>> = {
    [Key in keyof Obj as Uppercase<Key & string>]: Obj[Key]
}

// ToReadonly 索引类型的索引可以添加 readonly 的修饰符，代表只读
type ToReadonly<T> = {
  readonly [Key in keyof T]: T[Key]
}

// ToMutable 去掉readonly
type ToMutable<T> = {
  -readonly [Key in keyof T]: T[Key]
}

// ToRequired 去掉可选操作服
type ToRequired<T> = {
  [Key in keyof T]-?: T[Key]
}

// FilterByValueType 过滤
type FilterByValueType<Obj extends Record<string, any>, ValueType> = {
    [Key in keyof Obj as Obj[Key] extends ValueType ? Key : never]: Obj[Key]
}
interface Person {
    name: string;
    age: number;
    hobby: string[];
}
type res = FilterByValueType<Person, string | number>

递归复用 #

  • 递归是把问题分解为一系列相似的小问题，通过函数不断调用自身来解决这一个个小问题，直到满足结束条件，就完成了问题的求解
  • TypeScript 类型系统不支持循环，但支持递归。当处理数量（个数、长度、层数）不固定的类型的时候，可以只处理一个类型，然后递归的调用自身处理下一个类型，直到结束条件也就是所有的类型都处理完了，就完成了不确定数量的类型编程，达到循环的效果
  • Function不是索引类型
  • interface 主要是声明索引类型的，type 可以声明任何类型
typescript
// Promise的递归调用
// DeepPromiseValueType 实现提取不确定层数的Promise中的Value
type DeepPromiseValueType<T extends Promise<unknown>> = T extends Promise<infer ValueType> ? ValueType extends Promise<unknown> ? DeepPromiseValueType<ValueType> : ValueType : never
// 简易写法
type DeepPromiseValueType<T> = T extends Promise<infer ValueType> ? 
 DeepPromiseValueType<ValueType> : T
type ttt = DeepPromiseValueType<Promise<Promise<Promise<number>>>>


// 数组类型的递归
// ReverseArr 反转数组
type ReverseArr<Arr extends any[]> = Arr extends [...infer Rest, infer First] ? [First, ...ReverseArr<Rest>] : []
type ReverseArr<Arr extends unknown[]> = Arr extends [
    infer First,
    ...infer Center,
    infer End
]
    ? [End, ...ReverseArr<Center>, First]
    : Arr;
type res = ReverseArr<[1, 2, 3, 4, 5]>

// includes 查找相符的类型
type IncludesArr<Arr extends any[], Target> = Arr extends [infer First, ...infer Rest] ? IsEqual<First, Target> extends true ? true
    : IncludesArr<Rest, Target> : false
// 相等的判断就是 A 是 B 的子类型并且 B 也是 A 的子类型
type IsEqual<T, R> = (T extends R ? true : false) & (R extends T ? true : false)
type res = IncludesArr<[1, 2, 3, 4, 5], 5>

// RemoveItem 删除目标类型，返回一个新的数组类型
// Result是构造出的新数组
type RemoveItem<Arr extends unknown[], Item, Result extends unknown[] = []> = Arr extends [infer First, ...infer Rest] ? isEqual<First, Item> extends true ? RemoveItem<Rest, Item, Result> : RemoveItem<Rest, Item, [...Result, First]> : Result
type res = RemoveItem<[1, 2, 3, 4], 3>

// BuildArray 构造定长的数组
// 每次判断Arr的长度是否到了Length
type BuildType<Length extends number, Ele = unknown, Arr extends unknown[] = []> = Arr['length'] extends Length
    ? Arr : BuildType<Length, Ele, [...Arr, Ele]>
type res = BuildType<6, 1>


// 字符串类型的递归
// ReplaceAll 替换多个目标字符
type ReplaceAll<Str extends string, From extends string, To extends string> =
    Str extends `${infer Left}${From}${infer Right}` ? `${Left}${To}${ReplaceAll<Right, From, To>}` : Str
type res = ReplaceAll<'abbc', 'b', 'c'>

// StringToUnion 将字符串的每个字符都提取出来组成联合类型
type StringToUnion<Str extends string> = Str extends `${infer First}${infer Rest}` ? First | StringToUnion<Rest> : never
type res = StringToUnion<'abc'>

// ReverseStr 字符串的反转
type ReverseStr<Str extends string, Result extends string = ''> = Str extends `${infer First}${infer Rest}` ? ReverseStr<Rest, `${First}${Result}`> : Result
type ReverseStr<Str extends string> = Str extends `${infer First}${infer Rest}` ? `${ReverseStr<Rest>}${First}` : Str
type res = ReverseStr<'abc'>


// 对象类型的递归
// DeepReadonly
// ts只有整个类型被用到的时候才会做类型计算，所以看起来只有最外层有readonly
type DeepReadonly<Obj extends Record<string, any>> = {
    readonly [Key in keyof Obj]:
        Obj[Key] extends object
            ? Obj[Key] extends Function
                ? Obj[Key] 
                : DeepReadonly<Obj[Key]>
            : Obj[Key]
}
// 在前面加上一段 Obj extends never ? never 或者 Obj extends any 等，触发计算
// Obj extends any 还有额外的好处就是能处理联合类型
type DeepReadonly<Obj extends Record<string, any>> =
    Obj extends any
        ? {
            readonly [Key in keyof Obj]:
                Obj[Key] extends object
                    ? Obj[Key] extends Function
                        ? Obj[Key] 
                        : DeepReadonly<Obj[Key]>
                    : Obj[Key]
        }
        : never;
